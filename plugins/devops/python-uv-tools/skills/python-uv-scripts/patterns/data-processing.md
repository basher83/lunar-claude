## Quick Example

```python
#!/usr/bin/env -S uv run
# /// script
# requires-python = ">=3.11"
# dependencies = ["polars>=0.20.0"]
# ///
import polars as pl

def analyze_csv(file_path: str):
    df = pl.read_csv(file_path)

    # Basic analysis
    summary = df.describe()
    print(summary)

    # Filter and aggregate
    result = (
        df.filter(pl.col("value") > 100)
          .groupby("category")
          .agg(pl.col("value").mean())
    )
    print(result)
```
